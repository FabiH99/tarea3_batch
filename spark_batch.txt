from pyspark.sql import SparkSession
from pyspark.sql.functions import col

# Crear sesión de Spark
spark = SparkSession.builder \
    .appName("BatchTickets") \
    .getOrCreate()

# Ruta del archivo
file_path = "tickets.csv"

# Leer archivo CSV
df = spark.read \
    .option("header", True) \
    .option("inferSchema", True) \
    .csv(file_path)

# Mostrar esquema
print("=== ESQUEMA ===")
df.printSchema()

# Mostrar datos
print("=== PRIMEROS REGISTROS ===")
df.show(5)

# Mostrar columnas
print("=== COLUMNAS ===")
print(df.columns)

# Contar registros
print("=== TOTAL DE REGISTROS ===")
total = df.count()
print(total)

# Agrupar por categoría
print("=== TICKETS POR CATEGORIA ===")
df.groupBy("categoria") \
  .count() \
  .orderBy(col("count").desc()) \
  .show(truncate=False)

# Detener Spark
spark.stop()
